Interactive plot

fruit_plot = Plot.plot({
  x: {label: "Fruit"},
  y: {label: "Weight"},
  marks: [
    Plot.barY(fruit_data, {
      x: "fruit", y: "weight", fill: "#999999"
    }),
    Plot.barY(fruit_data, {
      x: "fruit", y: "weight", fill: "#ff0000",
      filter: (d) => myfruit == d.fruit
    }),
    Plot.barY(fruit_data, Plot.pointer({
      x: "fruit", y: "weight", opacity: 0
    }))
  ]
});
